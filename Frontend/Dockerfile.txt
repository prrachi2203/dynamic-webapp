5) Monitor CPU usage with conditions
#!/bin/bash


iterations=$1
i=1


while [ $i -le $iterations ]
do
  cpu=$(top -bn1 | grep "Cpu(s)" | awk '{print 100 - $8}')


  echo "Iteration $i: CPU Usage = $cpu%"


  if (( $(echo "$cpu < 10" | bc -l) ))
  then
    ((i++))
    continue
  fi


  if (( $(echo "$cpu > 90" | bc -l) ))
  then
    echo "CPU exceeded 90%. Exiting."
    break
  fi


  echo "Logged CPU usage: $cpu%"
  ((i++))
done

7) Check if a number is prime using a function
#!/bin/bash


is_prime() {
  num=$1


  if [ "$num" -le 1 ]; then
    return 1
  fi


  for ((i=2; i<=num/2; i++))
  do
    if (( num % i == 0 ))
    then
      return 1
    fi
  done
  return 0
}


read -p "Enter a number: " n


if is_prime "$n"
then
  echo "$n is a prime number"
else
  echo "$n is not a prime number"
fi

#!/bin/bash


FILE="uptime_data.tsv"


touch "$FILE"


while true
do
  TIMESTAMP=$(date "+%Y-%m-%d %H:%M:%S")
  UPTIME=$(uptime -p)


  echo -e "$TIMESTAMP\t$UPTIME" >> "$FILE"


  # Sort in descending order (latest first)
  sort -r "$FILE" -o "$FILE"


  sleep 300   # 5 minutes
done

2) TCP Client-Server (Python)
Server (EC2)
import socket


HOST = '0.0.0.0'
PORT = 5000


server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind((HOST, PORT))
server.listen(1)


print("Server listening...")


while True:
    conn, addr = server.accept()
    print(f"Connected by {addr}")


    with open("uptime_data.tsv", "r") as f:
        latest = f.readline()


    conn.sendall(latest.encode())
    conn.close()

    cidr_blocks = ["0.0.0.0/0"]

  ingress {
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"

    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}


# EC2 Instance
resource "aws_instance" "web" {
  ami           = "ami-00bb6a80f"
  instance_type = "t2.micro"
  key_name      = "mykey"


  security_groups = [aws_security_group.web_sg.name]


  root_block_device {
    volume_size = 10

              #!/bin/bash

    Name = "WebAppInstance"
  }

Set C
2A & 2B) Using docker-compose.yml
docker-compose.yml
version: '3.8'


services:
  mongo:
    image: mongo
    container_name: mongo-db
    restart: always
    environment:
      MONGO_INITDB_ROOT_USERNAME: admin
      MONGO_INITDB_ROOT_PASSWORD: admin123
    ports:
      - "27017:27017"


  mongo-express:
    image: mongo-express
    container_name: mongo-express
    restart: always
    depends_on:
      - mongo
    environment:
      ME_CONFIG_MONGODB_ADMINUSERNAME: admin
      ME_CONFIG_MONGODB_ADMINPASSWORD: admin123
      ME_CONFIG_MONGODB_SERVER: mongo
    ports:
      - "8081:8081"


________________


Run docker-compose
docker-compose up -d


________________


Stop containers
docker-compose down


________________


How it works
* Docker Compose creates a default network automatically
* Services communicate using service names (mongo)
* depends_on ensures Mongo starts before mongo-express
* Browser access:
http://localhost:8081
